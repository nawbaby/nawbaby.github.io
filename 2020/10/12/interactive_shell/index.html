脚本模板化

  • 解决需要将某些步骤的脚本重复粘贴到shell中执行
  • 一些操作是有步骤的并且都是可以用命令执行的
  • 不用翻笔记然后复制粘贴脚本执行
脚本demo
#!/bin/bash
stty erase '^H'
function_title ()
{
echo 
"
##磁带库操作命令集成脚本　       —author guoyanping  
##选择底下的操作序号
##***基础环境设置
## １、setenforce 0              ## 关闭 selinux
## ２、service iptables stop     ##关闭防火墙
## ３、service mhvtl start       ##启动磁带库软件，这样 lscssi -g 才能显示那些磁带
## ４、service tgtd  start
##***配置磁带库
## ５、lsscsi -g                 ## 显示虚拟磁带库中的对应的磁盘符号便于编辑tgt共享设置
## ６、vi /etc/tgt/targets.conf  ##修改配置文件
## ７、service tgtd reload       ## 修改配置文件是否成功完全看这个，stop，start  无法替代。
## ８、tgtadm --lld iscsi --mode target --op show  ## tgtd 与mhvtl合作 将target共享出去。
##***使用帮助
## ０、看下这个操作栏
## -1、退出
## 也可直接输入你想执行的命令 
"
}

function_title
while true
do
read  -p "输入需要进行的操作:" choice
case $choice in

	1)
	setenforce 0 
        getenforce
	;;
	2)
	service iptables stop  
	;;
	3)
	service mhvtl start 
	;;
	4)
	service tgtd  start 
	;;
	
	5)
	lsscsi -g 
	;;
	6)
	vi /etc/tgt/targets.conf
	;;
	7)
	service tgtd reload 
	;;
	8)
	tgtadm --lld iscsi --mode target --op show
	;;
	
    0)
    function_title
    ;;
	-1)
	exit
	;;
	*)
	$choice
	;;
esac
done
